defined_text = {
	name = hoi4pc_get_close

	text = {
		trigger = {
			OR = {
				any_allied_country = {
					has_war_with = losing_country^peace_gui_index
				}
				has_war_with = losing_country^peace_gui_index
			}
		}
		localization_key = peace_capitulation_close_good
	}
	text = {
		trigger = {
			OR = {
				tag = losing_country^peace_gui_index
				has_war_together_with = losing_country^peace_gui_index
				is_in_faction_with = losing_country^peace_gui_index
			}
		}
		localization_key = peace_capitulation_close_bad
	}
	text = {
		trigger = {
			NOT = { has_war_together_with = losing_country^peace_gui_index }
			NOT = { is_in_faction_with = losing_country^peace_gui_index }
			NOT = { any_allied_country = { has_war_with = losing_country^peace_gui_index } }
			NOT = { has_war_with = losing_country^peace_gui_index }
			NOT = { tag = losing_country^peace_gui_index }
		}
		localization_key = peace_capitulation_close_neutral
	}
}
